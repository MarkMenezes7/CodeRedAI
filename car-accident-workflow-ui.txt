      <article class="card feed">

        <div class="event-list" id="hospitalFeed"></div>
      </article>

      <article class="card feed">
        <h3>Driver Dispatch Feed</h3>
        <p>Nearest driver receives crash location, route ETA, and patient contact.</p>
        <div class="event-list" id="driverFeed"></div>

  <script>
    const carModel = document.getElementById('carModel');
    const airbagAction = document.getElementById('airbagAction');
    const sendAlertBtn = document.getElementById('sendAlertBtn');
    const resetBtn = document.getElementById('resetBtn');
    const patientName = document.getElementById('patientName');
    const patientPhone = document.getElementById('patientPhone');
    const dispatchStatus = document.getElementById('dispatchStatus');
    const locationStatus = document.getElementById('locationStatus');
    const hospitalFeed = document.getElementById('hospitalFeed');
    const driverFeed = document.getElementById('driverFeed');
    const clock = document.getElementById('clock');
    const cursorGlow = document.getElementById('cursorGlow');

    const CAR_PRESET_DATA = {
      'Honda City': { patientName: 'Rohan Mehta', patientPhone: '+91 98765 11001' },
      'Maruti Suzuki Swift': { patientName: 'Neha Sharma', patientPhone: '+91 98765 11002' },
      'Hyundai i20': { patientName: 'Aman Patel', patientPhone: '+91 98765 11003' },
      'Tata Nexon': { patientName: 'Priya Nair', patientPhone: '+91 98765 11004' },
      'Mahindra XUV700': { patientName: 'Vikram Singh', patientPhone: '+91 98765 11005' },
      'Kia Seltos': { patientName: 'Isha Kapoor', patientPhone: '+91 98765 11006' },
      'Toyota Innova Hycross': { patientName: 'Sandeep Rao', patientPhone: '+91 98765 11007' },
      'Skoda Slavia': { patientName: 'Meera Joshi', patientPhone: '+91 98765 11008' },
      'Volkswagen Virtus': { patientName: 'Arjun Das', patientPhone: '+91 98765 11009' },
      'Renault Kiger': { patientName: 'Karan Bedi', patientPhone: '+91 98765 11010' }
    };

    const severityBands = ['High', 'Critical', 'Moderate'];
    const etaBands = ['4 min', '6 min', '8 min', '10 min'];
    let currentLocation = null;
    let isLocationLoading = false;

    function now() {
      return new Date().toLocaleString('en-IN', {
        hour12: true,
        day: '2-digit',
        month: 'short',
        year: 'numeric',
        hour: '2-digit',
        minute: '2-digit',
        second: '2-digit'
      });
    }

    function randomItem(items) {
      return items[Math.floor(Math.random() * items.length)];
    }

    function updateClock() {
      clock.textContent = now();
    }

    function setStatus(message, alertMode) {
      dispatchStatus.textContent = message;
      dispatchStatus.className = alertMode ? 'status alert' : 'status';
    }

    function setLocationMessage(message, type) {
      locationStatus.textContent = message;
      if (type === 'error') {
        locationStatus.className = 'status alert';
        return;
      }

      if (type === 'ready') {
        locationStatus.className = 'status good';
        return;
      }

      if (type === 'loading') {
        locationStatus.className = 'status loading';
        return;
      }

      locationStatus.className = 'status';
    }

    function triggerDispatchBurst(button) {
      for (let i = 0; i < 10; i += 1) {
        const spark = document.createElement('span');
        spark.className = 'spark';
        spark.style.setProperty('--dx', (Math.random() * 120 - 60).toFixed(1) + 'px');
        spark.style.setProperty('--dy', (Math.random() * 90 - 45).toFixed(1) + 'px');
        button.appendChild(spark);
        setTimeout(function () {
          spark.remove();
        }, 700);
      }
    }

    function addEvent(target, title, meta, tagClass, tagLabel) {
      const node = document.createElement('div');
      node.className = 'event';
      node.innerHTML = [
        '<strong>' + title + '</strong>',
        '<span class="meta">' + meta + '</span>',
        '<span class="tag ' + tagClass + '">' + tagLabel + '</span>'
      ].join('');
      target.prepend(node);
    }

    function canSend() {
      return Boolean(carModel.value) &&
        airbagAction.value === 'activate' &&
        Boolean(currentLocation) &&
        !isLocationLoading;
    }

    function refreshButtons() {
      sendAlertBtn.disabled = !canSend();
    }

    function requestCurrentLocation() {
      if (!navigator.geolocation) {
        currentLocation = null;
        isLocationLoading = false;
        setLocationMessage('Geolocation is not supported in this browser.', 'error');
        refreshButtons();
        return;
      }

      currentLocation = null;
      isLocationLoading = true;
      refreshButtons();
      setLocationMessage('Capturing current location. Please allow location access.', 'loading');

      navigator.geolocation.getCurrentPosition(
        function (position) {
          currentLocation = {
            lat: position.coords.latitude.toFixed(6),
            lng: position.coords.longitude.toFixed(6)
          };
          isLocationLoading = false;
          setLocationMessage(
            'Location captured: Lat ' + currentLocation.lat + ', Lng ' + currentLocation.lng,
            'ready'
          );
          refreshButtons();
        },
        function (error) {
          currentLocation = null;
          isLocationLoading = false;

          if (error && error.code === error.PERMISSION_DENIED) {
            setLocationMessage('Location access denied. Please allow permission and try Activate airbags again.', 'error');
          } else if (error && error.code === error.TIMEOUT) {
            setLocationMessage('Location request timed out. Try Activate airbags again.', 'error');
          } else {
            setLocationMessage('Could not capture current location. Please try again.', 'error');
          }

          refreshButtons();
        },
        {
          enableHighAccuracy: true,
          timeout: 10000,
          maximumAge: 0
        }
      );
    }

    carModel.addEventListener('change', function () {
      const enabled = Boolean(carModel.value);
      airbagAction.disabled = !enabled;
      currentLocation = null;
      isLocationLoading = false;

      if (!enabled) {
        airbagAction.value = '';
        patientName.value = '';
        patientPhone.value = '';
        setLocationMessage('Location feed idle. GPS coordinates will appear after alert.', 'ready');
      } else {
        const profile = CAR_PRESET_DATA[carModel.value] || {
          patientName: 'Unknown Patient',
          patientPhone: 'Not Provided'
        };
        patientName.value = profile.patientName;
        patientPhone.value = profile.patientPhone;
        setLocationMessage('Location pending. Select Activate airbags to capture current GPS.', 'loading');
      }

      refreshButtons();
      setStatus(
        enabled
          ? 'Vehicle selected: ' + carModel.value + '. Patient details prefilled. Choose Activate airbags to capture location.'
          : 'Waiting for action. Select a car model, then choose Activate airbags.',
        false
      );
    });

    airbagAction.addEventListener('change', function () {
      if (airbagAction.value === 'activate') {
        setStatus('Airbag command armed. Capturing your current location now.', false);
        requestCurrentLocation();
      } else {
        currentLocation = null;
        isLocationLoading = false;
        setLocationMessage('Location pending. Select Activate airbags to capture current GPS.', 'loading');
        refreshButtons();
      }
    });

    sendAlertBtn.addEventListener('click', function () {
      if (!canSend()) return;

      const location = currentLocation;
      const severity = randomItem(severityBands);
      const eta = randomItem(etaBands);
      const name = patientName.value.trim() || 'Unknown Patient';
      const phone = patientPhone.value.trim() || 'Not Provided';
      const stamp = now();

      triggerDispatchBurst(sendAlertBtn);

      setStatus(
        'Emergency alert dispatched. Hospital and nearest driver notified for ' + carModel.value + '.',
        true
      );

      locationStatus.className = 'status good';
      locationStatus.textContent =
        'GPS locked: Lat ' + location.lat + ', Lng ' + location.lng + ' | Severity: ' + severity;

      addEvent(
        hospitalFeed,
        'Crash Alert: ' + carModel.value,
        'Patient: ' + name + ' | Contact: ' + phone + ' | Location: ' + location.lat + ', ' + location.lng + ' | Severity: ' + severity + ' | ' + stamp,
        'tag-hospital',
        'Hospital'
      );

      addEvent(
        driverFeed,
        'Dispatch: Proceed to Accident Spot',
        'Vehicle: ' + carModel.value + ' | Patient: ' + name + ' | Coordinates: ' + location.lat + ', ' + location.lng + ' | ETA: ' + eta + ' | ' + stamp,
        'tag-driver',
        'Driver'
      );
    });

    resetBtn.addEventListener('click', function () {
      carModel.value = '';
      airbagAction.value = '';
      airbagAction.disabled = true;
      patientName.value = '';
      patientPhone.value = '';
      currentLocation = null;
      isLocationLoading = false;
      hospitalFeed.innerHTML = '';
      driverFeed.innerHTML = '';
      setStatus('Waiting for action. Select a car model, then choose Activate airbags.', false);
      setLocationMessage('Location feed idle. GPS coordinates will appear after alert.', 'ready');
      refreshButtons();
    });

    updateClock();
    setInterval(updateClock, 1000);

    if (cursorGlow) {
      window.addEventListener('pointermove', function (event) {
        cursorGlow.style.opacity = '1';
        cursorGlow.style.transform =
          'translate(' + (event.clientX - 160) + 'px, ' + (event.clientY - 160) + 'px)';
      });

      window.addEventListener('pointerleave', function () {
        cursorGlow.style.opacity = '0';
      });
    }
  </script>
</body>
